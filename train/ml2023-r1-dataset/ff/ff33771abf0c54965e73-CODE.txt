csharp using UnityEngine; public class AI : MonoBehaviour { public Transform target; public float speed = 5f; private void Update() { // Проверяем, есть ли цель if (target != null) { // Поворачиваемся к цели Vector3 direction = target.position - transform.position; Quaternion rotation = Quaternion.LookRotation(direction); transform.rotation = Quaternion.Lerp(transform.rotation, rotation, Time.deltaTime * speed); // Двигаемся вперед transform.Translate(Vector3.forward * Time.deltaTime * speed); } } }
