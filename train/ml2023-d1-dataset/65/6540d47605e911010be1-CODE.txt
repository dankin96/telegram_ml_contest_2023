# 输入的表格
table = [['张三', 18, '男'],
         ['李四', 17, '女'],
         ['王五', 16, '男'],
         ['赵丽颖', 20, '女'],
         ['刘亦菲', 19, '女'],
         ['张二', 21, '男'],
         ['陈小明', 18, '男'],
         ['杨华', 17, '女']]

# 要删除的名字列表，注意列表中的名字要与表格中的一致
remove_names = ['张三', '李四', '王五', '张二', '赵丽颖', '刘亦菲']

# 删除指定名字后的表格
new_table = [row for row in table if row[0] not in remove_names]

# 输出新表格
for row in new_table:
    print(row)
